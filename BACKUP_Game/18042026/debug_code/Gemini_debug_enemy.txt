// enemy.js - Enemy creation and update logic
window.enemies = window.enemies || [];
const enemies = window.enemies;
let currentWave = 1;
let waveEnemyCount = 5;
let nextWaveCountdown = 0;
const waveDisplay = document.getElementById('waveDisplay');

// UI and Score Tracking
window.score = window.score || 0;
window.killCount = window.killCount || 0;
const scoreDisplay = document.getElementById('scoreDisplay');
const killCountDisplay = document.getElementById('killCountDisplay');

function createEnemyWave() {
    // CRITICAL FIX: Ensure scene and camera exist before spawning
    if (typeof scene === 'undefined' || typeof camera === 'undefined') return;

    for (let i = 0; i < waveEnemyCount; i++) {
        const enemyGeometry = new THREE.BoxGeometry(1, 2, 1);
        const enemyMaterial = new THREE.MeshBasicMaterial({ color: 0x33ff66 });
        const enemy = new THREE.Mesh(enemyGeometry, enemyMaterial);
        
        // Spawn in a circle around the player
        const angle = Math.random() * Math.PI * 2;
        const distance = 10 + Math.random() * 10;
        enemy.position.set(
            camera.position.x + Math.cos(angle) * distance,
            1, // Height of the enemy
            camera.position.z + Math.sin(angle) * distance
        );
        
        enemy.userData = {
            health: 1,
            speed: 0.05 + (currentWave * 0.01),
            dropsAmmo: true
        };
        
        scene.add(enemy);
        enemies.push(enemy);
    }
}

function updateEnemies() {
    if (window.gameOver) return;
    
    // 1. Move enemies toward the player
    for (let i = enemies.length - 1; i >= 0; i--) {
        const enemy = enemies[i];
        const direction = new THREE.Vector3().subVectors(camera.position, enemy.position);
        direction.y = 0; // Stay on the ground
        direction.normalize();
        enemy.position.add(direction.multiplyScalar(enemy.userData.speed));

        // 2. Check if enemy died
        if (enemy.userData.health <= 0) {
            if (typeof createAmmoDropAt === 'function') {
                createAmmoDropAt(enemy.position);
            }
            window.score += 100;
            window.killCount += 1;
            if (scoreDisplay) scoreDisplay.textContent = `Score: ${window.score}`;
            if (killCountDisplay) killCountDisplay.textContent = `Kills: ${window.killCount}`;

            scene.remove(enemy);
            enemies.splice(i, 1);
        }
    }
    
    // 3. Wave Management: Spawn next wave if all enemies are dead
    if (enemies.length === 0 && nextWaveCountdown === 0) {
        nextWaveCountdown = 120; // 2 second delay at 60fps
    }
    
    if (nextWaveCountdown > 0) {
        nextWaveCountdown--;
        if (nextWaveCountdown === 0) {
            currentWave++;
            waveEnemyCount = 5 + (currentWave * 2);
            if (waveDisplay) waveDisplay.textContent = `Wave: ${currentWave}`;
            createEnemyWave();
        }
    }
}

// Ensure first wave spawns once the game starts
function initEnemies() {
    if (typeof scene !== 'undefined' && scene) {
        createEnemyWave();
    } else {
        setTimeout(initEnemies, 100);
    }
}

window.updateEnemies = updateEnemies;
initEnemies();
